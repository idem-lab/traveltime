
%%
%% The "author" command and its associated commands are used to define
%% the authors and their affiliations.
%% Of note is the shared affiliation of the first two authors, and the
%% "authornote" and "authornotemark" commands
%% used to denote shared contribution to the research.

$for(by-author)$
  \author{$by-author.name.literal$}
  $if(by-author.orcid)$\orcid{$by-author.orcid$}$endif$
  $if(by-author.affiliations)$
    $for(by-author.affiliations)$
      \affiliation{%
        $if(by-author.affiliations.name)$
          \institution{$by-author.affiliations.name$}
        $endif$
        $if(by-author.affiliations.address)$
          \streetaddress{$by-author.affiliations.address$}
        $endif$
        $if(by-author.affiliations.city)$
          \city{$by-author.affiliations.city$}
        $endif$
        $if(by-author.affiliations.state)$
          \state{$by-author.affiliations.state$}
        $endif$
        $if(by-author.affiliations.country)$
          \country{$by-author.affiliations.country$}
        $endif$
        $if(by-author.affiliations.postal-code)$
          \postcode{$by-author.affiliations.postal-code$}
        $endif$
      }
    $endfor$
  $endif$
$endfor$

$if(acm-metadata.shortauthors)$
\renewcommand{\shortauthors}{$acm-metadata.shortauthors$}
$endif$
